20/**
21 * Generates human-readable titles for ability models.
22 *
23 * This class analyzes ability configurations and produces descriptive titles
24 * based on the ability's scope, target entity, and ownership constraints. It
25 * handles wildcard abilities, model-specific permissions, and various permission
26 * patterns to create user-friendly descriptions for display in admin interfaces.
27 *

32    /**
33     * Create a new ability title generator for the given ability model.
34     *
35     * Analyzes the ability's configuration (name, subject_type, subject_id, only_owned)
36     * to determine the appropriate human-readable title. Handles multiple permission
37     * patterns including wildcards, simple abilities, model-specific permissions,
38     * and ownership-based restrictions.
39     *

58        } elseif ($this->isSpecificModelAbility($ability)) {
59            $this->title = $this->getSpecificModelAbilityTitle($ability);
60        } elseif ($this->isGlobalActionAbility($ability)) {
